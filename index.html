07.轮播默认样式

代码展示：

                     
                     //引入swiper.css, swiper-bundle.min.css控制样式
                     //引入swiper-bundle.min.js控制JS
                            
                     <div class="swiper margin-bottom-md"> 
                        <div class="swiper-wrapper">
                            <div class="swiper-slide bg swiper-bg-sizing">
                            </div>
                            <div class="swiper-slide bg swiper-bg-sizing">
                            </div>
                            <div class="swiper-slide bg swiper-bg-sizing">
                            </div>
                        </div>
                        <!-- 分页器 -->
                        <div class="swiper-pagination"></div>
                        <!-- 导航按钮 -->
                        <div class="swiper-button-prev nav-hover"></div>
                        <div class="swiper-button-next nav-hover"></div>
                    </div>

                    
                    <!-- JS初始化 -->
                    <script>

                    var mySwiper = new Swiper('.swiper', {
                        direction: 'horizontal', // 水平轮播
                        loop: true, // 循环模式选项
                        //分页器
                        pagination: {
                            el: '.swiper-pagination',
                        },
                        //前进后退按钮
                        navigation: {
                            nextEl: '.swiper-button-next',
                            prevEl: '.swiper-button-prev',
                        },
                        //自动播放
                        autoplay : true,
                        autoplay:{
                            delay: 5000, //自动播放速度
                            disableOnInteraction : false, //用户操作后是否停止自动播放
                        },
                        speed : 1000,//轮播切换速度
                    }) 

                    </script>
